    Usage

    onClick ="$('#tableID').tableExport({type:'pdf',escape:'false'});"

    Options

    separator: ','
    ignoreColumn: [2,3],
    tableName:'yourTableName'
    type:'csv'
    pdfFontSize:14
    pdfLeftMargin:20
    escape:'true'
    htmlContent:'false'
    consoleLog:'false'
